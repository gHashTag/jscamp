  // –∏–Ω–∏—Ü–∏–∏—Ä—É–µ–º –º–∞—Å—Å–∏–≤ –æ–±—ä–µ–∫—Ç–æ–≤ ToDo –ª–∏—Å—Ç–∞ 
  todos = [
    { id: '01', title: 'Hello', completed: false },
    { id: '02', title: 'Hello2', completed: false }
  ]

  // makeAutoObservable –¥–µ–ª–∞–µ—Ç –≤—Å–µ —Å–≤–æ–π—Å—Ç–≤–∞ –Ω–∞–±–ª—é–¥–∞–µ–º—ã–º–∏ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é
  constructor() {
    makeAutoObservable(this)
  }

  // —Å—Ç–æ—Ä –≤ mobx –º—É—Ç–∞–±–µ–ª—å–Ω—ã–π, –ø–æ—ç—Ç–æ–º—É –ø—Ä–æ—Å—Ç–æ –ø—É—à–∏–º –≤ –Ω–µ–≥–æ –Ω–æ–≤—É—é –∑–∞–¥–∞—á—É
  createTodo(todo) {
    console.log(`create`)
    this.todos.push(todo)
  }

  // —É–¥–∞–ª—è–µ–º –ø–æ id –∑–∞–¥–∞—á—É, –æ—Ç—Ñ–∏–ª—å—Ç—Ä–æ–≤–∞–Ω–Ω—ã–π –º–∞—Å—Å–∏–≤ –ø–æ id 
  deleteTodo(id) {
    console.log(`delete`)
    this.todos = this.todos.filter(todo => todo.id !== id)
  }

  // —Å—Ç–∞–≤–∏–º  true completed –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω–æ–π –∑–∞–¥–∞—á–µ –∏–ª–∏ false –≤ –ø—Ä–æ—Ç–∏–≤–Ω–æ–º —Å–ª—É—á–∞–µ 
  completeTodo(id) {
    console.log(`complete`)
    this.todos = this.todos.map(todo => (todo.id === id ? { ...todo, completed: !todo.completed } : todo))
  }

}

export default new Todo()
